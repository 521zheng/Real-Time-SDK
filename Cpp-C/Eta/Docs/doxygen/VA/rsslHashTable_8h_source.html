00150 {
00151     RsslUInt32 newQueueCount = pTable->queueCount * 2 + 1;
00152         RsslUInt64 queueListSize = newQueueCount * sizeof(RsslQueue);
00153     RsslQueue* rsslOldQueue = (RsslQueue*)pTable->queueList;
00154         RsslUInt32 i;

00157         pTable->thresholdCapacity = (RsslUInt32)((double)newQueueCount * LOAD_FACTOR);

00159         if (queueListSize > (size_t)-1) /* overflow */
00160                 return RSSL_RET_FAILURE;

00164 
00165         if (!pTable->queueList)
00166                 return RSSL_RET_FAILURE;

00169         for (i = 0; i < newQueueCount; ++i)
00170         {
00171                 rsslInitQueue(&pTable->queueList[i]);
00172         }
00173 
00174     /* iterate through old hash table to move all the items */
00175         for (i = 0; i < pTable->queueCount; ++i)
00176         {
00177         RsslQueueLink* pQueueLink = rsslQueueRemoveFirstLink(&rsslOldQueue[i]);
00178         while (pQueueLink)

00180             RsslQueue* pQueue;
00181             RsslHashLink *pHashLink = RSSL_HASH_LINK_FROM_QUEUE_LINK(pQueueLink);
00182             RsslUInt32 queueListLocation;

00186                 pQueue = &pTable->queueList[queueListLocation];
