00328 static RsslRDMMsg *rsslCreateRDMMsgCopy(RsslRDMMsg *pRdmMsg, RsslUInt32 lengthHint, RsslRet *pRet)
00329 {
00330     RsslUInt32 msgSize = sizeof(RsslRDMMsg);
00331 
00332     if (lengthHint == 0)
00333         lengthHint = 1;
00334 
00335     /* Copy the message, resizing if necessary. */
00336     while (1)
00337     {
00338         RsslRDMMsg *pNewRdmMsg;
00339         RsslRet ret;
00340         char *pData;
00341         RsslBuffer memoryBuffer;
00342 
00343         if ((pData = (char*)malloc(msgSize + lengthHint)) == NULL)
00344         {
00345             *pRet = RSSL_RET_BUFFER_NO_BUFFERS;
00346             return NULL;
00347         }
00348 
00349         /* Message struct goes first. */
00350         pNewRdmMsg = (RsslRDMMsg*)pData;
00351 
00352         /* Remaining memory is used for the deep copy. */
00353         memoryBuffer.data = pData + msgSize;
00354         memoryBuffer.length = lengthHint;
00355 
00356         ret = rsslCopyRDMMsg(pNewRdmMsg, pRdmMsg, &memoryBuffer);
00357 
00358         switch(ret)
00359         {
00360             case RSSL_RET_SUCCESS:
00361                 *pRet = RSSL_RET_SUCCESS;
00362                 return pNewRdmMsg;
00363 
00364             case RSSL_RET_BUFFER_TOO_SMALL:
00365                 free(pData);
00366                 lengthHint *= 2;
00367                 continue;
00368 
00369             default:
00370                 free(pData);
00371                 *pRet = RSSL_RET_FAILURE;
00372                 return NULL;
00373         }
00374     }
