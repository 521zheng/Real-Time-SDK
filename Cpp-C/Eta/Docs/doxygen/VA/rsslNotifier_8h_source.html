00011  /* Cross-platform notification.
00012   * 
00013   * Overview of usage:
00014   * - rsslCreateNotifier creates an RsslNotifier that can wait on file descriptors.
00015   * - rsslCreateNotifierEvent creates an RsslNotifierEvent, associated with a file descriptor.
00016   * - rsslNotifierAddEvent adds the RsslNotifierEvent to the RsslNotifier.
00017   * - rsslNotifierRegisterRead/rsslNotifierRegisterWrite enable read/write notification for the RsslNotifierEvent.
00018   * - rsslNotifierWait waits for notification on the RsslNotifierEvents associated with the RsslNotifier, for the specified time.
00019   *     The triggered events can be found in the RsslNotifier.notifiedEvents array. 
00020   *     They will remain there till the next call to rsslNotifierWait.
00021   * - rsslNotifierUnregisterRead/rsslNotifierUnregisterWrite disable read/write notification for the RsslNotifierEvent.
00022   * - rsslNotifierRemoveEvent removes an RsslNotifierEvent from the RsslNotifier.
00023   * - rsslDestroyNotifierEvent cleans up an RsslNotifierEvent.
00024   * - rsslDestroyNotifier cleans up the RsslNotifier. */

00037 /* Flags for notifier events. */
00038 typedef enum
00039 {
00040     RSSL_NESF_READ = 0x1,   /* Indicates file can be read from without blocking. */
00041     RSSL_NESF_WRITE = 0x2,  /* Indicates file can be written to without blocking. */
00042     RSSL_NESF_BAD_FD = 0x4  /* Indicates descriptor may be invalid. */
00043 } RsslNotifierEventFlags;
00044 
00045 typedef struct 
00046 {
00047     int notifiedFlags; /* Flags indicating whether read or write would block. See RsslNotifierEventFlags. */
00048 } RsslNotifierEvent;
00049 
00050 /* Indicates whether the event's file can be read from without blocking. */
00051 RTR_C_INLINE int rsslNotifierEventIsReadable(RsslNotifierEvent *pEvent)
00052 {
00053     return pEvent->notifiedFlags & RSSL_NESF_READ;
00054 }
00055 
00056 /* Indicates whether the event's file can be written to without blocking. */
00057 RTR_C_INLINE int rsslNotifierEventIsWritable(RsslNotifierEvent *pEvent)
00058 {
00059     return pEvent->notifiedFlags & RSSL_NESF_WRITE;
00060 }
00061 
00062 /* Indicates whether the event's file descriptor may be invalid.  The event may need its associated FD to be updated.
00063  *   Note: When the notifier uses select for notification, this will be set on every descriptor when
00064  *   it sees the EBADF error. When the notifier uses poll, it will be set only on appropriate events. */
00065 RTR_C_INLINE int rsslNotifierEventIsFdBad(RsslNotifierEvent *pEvent)
00066 {
00067     return pEvent->notifiedFlags & RSSL_NESF_BAD_FD;
00068 }
00069 
00070 /* Clears the event's notification flags. May be useful if you are done reading from/writing to a notified event. */
00071 RTR_C_INLINE void rsslNotifierEventClearNotifiedFlags(RsslNotifierEvent *pEvent)
00072 {
00073     pEvent->notifiedFlags = 0;
00074 }
