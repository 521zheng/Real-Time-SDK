00189 RTR_C_INLINE RsslRet rsslInitDefaultRDMLoginRequest(RsslRDMLoginRequest *pRequest, RsslInt32 streamId)
00190 {
00191     RsslUInt32 ipAddress = 0;
00192     char tmpHostNameBlock[256];
00193     RsslBuffer tmpHostName = { 256, tmpHostNameBlock };
00194 
00195     rsslClearRDMLoginRequest(pRequest);
00196 
00197     pRequest->rdmMsgBase.streamId = streamId;
00198 
00199     /* Set username to system name */
00200     pRequest->userName.data = pRequest->defaultUsername;
00201     pRequest->userName.length = sizeof(pRequest->defaultUsername);
00202     if (rsslGetUserName(&pRequest->userName) != RSSL_RET_SUCCESS)
00203         return RSSL_RET_FAILURE;
00204 
00205     /* Set application name to "upa" */
00206     pRequest->flags |= RDM_LG_RQF_HAS_APPLICATION_NAME;
00207     pRequest->applicationName.data = (char *)"upa";
00208     pRequest->applicationName.length = 3;
00209 
00210     /* Specify an application ID */
00211     pRequest->flags |= RDM_LG_RQF_HAS_APPLICATION_ID;
00212     pRequest->applicationId.data = (char *)"256"; 
00213     pRequest->applicationId.length = 3;
00214 
00215 
00216     /* Set position to current IP address. */
00217     pRequest->flags |= RDM_LG_RQF_HAS_POSITION;
00218 
00219     if (gethostname(tmpHostNameBlock, sizeof(tmpHostNameBlock)) != 0)
00220         snprintf(tmpHostNameBlock, 256, "localhost");
00221 
00222     if (rsslHostByName(&tmpHostName, &ipAddress) == RSSL_RET_SUCCESS)
00223     {
00224         rsslIPAddrUIntToString(ipAddress, tmpHostNameBlock);
00225         snprintf(pRequest->defaultPosition, sizeof(pRequest->defaultPosition), "%s/net", tmpHostNameBlock);
00226         pRequest->position.data = pRequest->defaultPosition;
00227         pRequest->position.length = (RsslUInt32)strlen(pRequest->defaultPosition);
00228     }
00229     else
00230     {
00231         snprintf(pRequest->defaultPosition, 256, "localhost");
00232         pRequest->position.data = pRequest->defaultPosition;
00233         pRequest->position.length = (RsslUInt32)strlen(pRequest->defaultPosition);
00234     }
00235     
00236     return RSSL_RET_SUCCESS;
00237 }
