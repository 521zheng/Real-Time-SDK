cmake_minimum_required (VERSION 3.8)

add_executable( IProvider130_static IProvider.cpp )
add_executable( IProvider130_shared IProvider.cpp )

if (WIN32)
  target_link_libraries( IProvider130_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( IProvider130_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( IProvider130_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( IProvider130_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( IProvider130_static PROPERTIES OUTPUT_NAME IProvider130 )
    target_link_libraries( IProvider130_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( IProvider130_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( IProvider130_shared PROPERTIES OUTPUT_NAME IProvider130 )
endif()
