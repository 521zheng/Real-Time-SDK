cmake_minimum_required (VERSION 3.8)

add_executable( NIProvider100_static NiProvider.cpp )
add_executable( NIProvider100_shared NiProvider.cpp )

if (WIN32)
  target_link_libraries( NIProvider100_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( NIProvider100_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( NIProvider100_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider100_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( NIProvider100_static PROPERTIES OUTPUT_NAME NIProvider100 )
    target_link_libraries( NIProvider100_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider100_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( NIProvider100_shared PROPERTIES OUTPUT_NAME NIProvider100 )
endif()
