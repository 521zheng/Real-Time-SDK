cmake_minimum_required (VERSION 3.8)

add_executable( NIProvider251_static NiProvider.cpp )
add_executable( NIProvider251_shared NiProvider.cpp )

if (WIN32)
  target_link_libraries( NIProvider251_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( NIProvider251_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( NIProvider251_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider251_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( NIProvider251_static PROPERTIES OUTPUT_NAME NIProvider251 )
    target_link_libraries( NIProvider251_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider251_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( NIProvider251_shared PROPERTIES OUTPUT_NAME NIProvider251 )
endif()
