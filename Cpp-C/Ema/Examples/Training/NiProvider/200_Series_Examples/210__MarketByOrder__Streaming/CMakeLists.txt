cmake_minimum_required (VERSION 3.8)

add_executable( NIProvider210_static NiProvider.cpp )
add_executable( NIProvider210_shared NiProvider.cpp )

if (WIN32)
  target_link_libraries( NIProvider210_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( NIProvider210_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( NIProvider210_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider210_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( NIProvider210_static PROPERTIES OUTPUT_NAME NIProvider210 )
    target_link_libraries( NIProvider210_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider210_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( NIProvider210_shared PROPERTIES OUTPUT_NAME NIProvider210 )
endif()
