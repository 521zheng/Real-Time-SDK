cmake_minimum_required (VERSION 3.8)

add_executable( NIProvider320_static NiProvider.cpp )
add_executable( NIProvider320_shared NiProvider.cpp )

if (WIN32)
  target_link_libraries( NIProvider320_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( NIProvider320_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( NIProvider320_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider320_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( NIProvider320_static PROPERTIES OUTPUT_NAME NIProvider320 )
    target_link_libraries( NIProvider320_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider320_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( NIProvider320_shared PROPERTIES OUTPUT_NAME NIProvider320 )
endif()
