cmake_minimum_required (VERSION 3.8)

add_executable( NIProvider300_static NiProvider.cpp )
add_executable( NIProvider300_shared NiProvider.cpp )

if (WIN32)
  target_link_libraries( NIProvider300_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( NIProvider300_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( NIProvider300_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider300_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( NIProvider300_static PROPERTIES OUTPUT_NAME NIProvider300 )
    target_link_libraries( NIProvider300_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( NIProvider300_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( NIProvider300_shared PROPERTIES OUTPUT_NAME NIProvider300 )
endif()
