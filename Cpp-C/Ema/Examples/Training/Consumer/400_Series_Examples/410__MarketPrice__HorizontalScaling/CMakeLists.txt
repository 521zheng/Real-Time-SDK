cmake_minimum_required (VERSION 3.8)

add_executable( Consumer410_static Consumer.cpp ConsumerManager.cpp )
add_executable( Consumer410_shared Consumer.cpp ConsumerManager.cpp )

if (WIN32)
  target_link_libraries( Consumer410_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( Consumer410_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( Consumer410_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( Consumer410_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( Consumer410_static PROPERTIES OUTPUT_NAME Consumer410 )
    target_link_libraries( Consumer410_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( Consumer410_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( Consumer410_shared PROPERTIES OUTPUT_NAME Consumer410 )
endif()
