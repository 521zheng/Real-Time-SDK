cmake_minimum_required (VERSION 3.8)

add_executable( Consumer100_static Consumer.cpp )
add_executable( Consumer100_shared Consumer.cpp )

if (WIN32)
  target_link_libraries( Consumer100_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( Consumer100_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( Consumer100_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( Consumer100_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( Consumer100_static PROPERTIES OUTPUT_NAME Consumer100 )
    target_link_libraries( Consumer100_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( Consumer100_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( Consumer100_shared PROPERTIES OUTPUT_NAME Consumer100 )
endif()
