cmake_minimum_required (VERSION 3.8)

add_executable( Consumer140_static Consumer.cpp )
add_executable( Consumer140_shared Consumer.cpp )

if (WIN32)
  target_link_libraries( Consumer140_static ${EXAMPLE_STATIC_LIBRARIES} )
  target_link_libraries( Consumer140_shared ${EXAMPLE_SHARED_LIBRARIES} )
else()
    target_link_libraries( Consumer140_static ${EXAMPLE_LIBRARIES_STATIC} ${SYSTEM_LIBRARIES} )
	set_target_properties( Consumer140_static PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR} )
	set_target_properties( Consumer140_static PROPERTIES OUTPUT_NAME Consumer140 )
    target_link_libraries( Consumer140_shared ${EXAMPLE_LIBRARIES_SHARED} ${SYSTEM_LIBRARIES} )
	set_target_properties( Consumer140_shared PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${EMA_EXAMPLES_OUTPUT_DIR}/Shared )
	set_target_properties( Consumer140_shared PROPERTIES OUTPUT_NAME Consumer140 )
endif()
